import { Injectable } from '@angular/core';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { environment } from '../../environments/environment';
import { RestApi } from "./restapi.service";

@Injectable()
export class ReportsService {

  endpoints: any;
  private sharedDTO: any;

  private isLoggedin: boolean = true;

  getisLoggedIn() {
    return this.isLoggedin;
  }



  constructor(private _http: HttpClient, private _restApi: RestApi) {
    this.endpoints = environment.endpoints;
  }


  setSharedDTO(val) {
    this.sharedDTO = val;
  }

  getSharedDTO() {
    return this.sharedDTO;
  }

  getLocaleLanguages(){
    return this._restApi.makeApiCall("get", this.endpoints.getLocaleLanguages);
  }

  getIncidents() {

  }

  getVehicleTrackData() {
    // return this._http.get<any>(this.endpoints.getVehicleTrackData);

    return this._restApi.makeApiCall("post", this.endpoints.getMaterialData, requestBody, { headers: this.getBuidingNameHeader("Kalyani Vista") });
  }

  getMaterialLogCount(category: string) {
    let url = this.endpoints.getMaterialLogCount.replace("{category}", category);
    return this._restApi.makeApiCall("get", url, undefined, { headers: this.getBuidingNameHeader("Kalyani Vista") });
  }

  getBuidingNameHeader(buildingName: string): HttpHeaders {
    let headers = new HttpHeaders({
      "X-BuildingName": buildingName
    });
    return headers;
  }

    let headers = this.getBuidingNameHeader('Kalyani Vista');
    // return this._http.post(this.endpoints.getMaterialReport, requestBody, { headers: headers, observe: 'response', responseType: 'blob' });
    return this._restApi.makeApiCall('post', this.endpoints.getMaterialReport, requestBody, { headers: headers, observe: 'response', responseType: 'blob' });
