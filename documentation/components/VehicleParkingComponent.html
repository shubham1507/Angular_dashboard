import { Component, OnInit, OnDestroy } from '@angular/core';
import { environment } from 'src/environments/environment';
import { ReportsService } from '../../../../services/reports.service';
import {
  Language,
  DefaultLocale,
  LocaleService,
  Timezone,
  TranslationService
} from 'angular-l10n';
import CalLocaleSwitch from 'src/app/shared/utils/cal-locale-switch';

@Component({
  selector: 'app-vehicle-parking',
  templateUrl: './vehicle-parking.component.html',
  styleUrls: ['./vehicle-parking.component.scss']
})
export class VehicleParkingComponent implements OnInit, OnDestroy {

  @Language() lang: string;
  @DefaultLocale() defaultLocale: string;
  @Timezone() timezone: string;
  calLocale: Object;
  issuesReportedText: string;
  overnightParkingText: string;
  poolParkingViolationText: string;
  vipParkingViolationText: string;
  /**
   * vehicle parking table data
   */
  vehTrackData: any;
  /**
   * vehicle parking overview data
   */
  vehicleParkingOverview: any;
  /**
   * parking occupancy donut chart data
   */
  parkOccList: Object[];
  /**
   * vehicle types donut chart data
   */
  vehTypeDistList: Object[];
  /**
   * date range to display the data
   */
  dateRange: Date[];

  //Adding the chart variables for parkOccConstants
  parkOccConstants_rightAlign: boolean = environment.parkOccConstants.CHARTPROPERTIES.RIGHTALIGN;;
  parkOccConstants_chartid: String = environment.parkOccConstants.CHARTPROPERTIES.CHARTID;
  parkOccConstants_addClearBtn: boolean = environment.parkOccConstants.CHARTPROPERTIES.ADDCLEARBTN;
  parkOccConstants_legendColors: any = environment.parkOccConstants.LEGENDCOLORS;
  parkOccConstants_options: any;

  // Adding the chart variable for the vehTypeDistrData
  vehTypeDistrData_rightAlign: boolean = environment.vehTypeDistrData.CHARTPROPERTIES.RIGHTALIGN;;
  vehTypeDistrData_chartid: String = environment.vehTypeDistrData.CHARTPROPERTIES.CHARTID;
  vehTypeDistrData_addClearBtn: boolean = environment.vehTypeDistrData.CHARTPROPERTIES.ADDCLEARBTN;
  vehTypeDistrData_legendColors: any = environment.vehTypeDistrData.LEGENDCOLORS;
  vehTypeDistrData_options: any;

  vehicleModal: any = {
    openvehicleModal: false,
  };

  vehicleDataSub: any;
  overviewSub: any;

  /**
   * Constructor function used to initialse the vehicle parking component
   * @param reportsService 
   */
  constructor(private reportsService: ReportsService, public locale: LocaleService, public translation: TranslationService) { }

  /**
   * shows vehicle report generating modal
   */
  openVehicleModal() {
    this.vehicleModal.openvehicleModal = true;
  }

  /**
   * Init function is used to fetch the data required for vehicle parking tab
   */
  ngOnInit() {
    this.parkOccConstants_options = environment.parkOccConstants.CHARTPROPERTIES;
    this.vehTypeDistrData_options = environment.vehTypeDistrData.CHARTPROPERTIES;
    this.parkOccList = [{ "key": "Parking Occupancy", "y": "1024" }, { "key": "Current Occupancy", "y": "842" }];
    this.vehTypeDistList = [{ "key": "Employee Vehicles - Cars", "y": "453" }, { "key": "Employee Vehicles - Bikes", "y": "345" }, { "key": "Employee Vehicles - Cycles", "y": "310" }];

    this.translation.translationChanged().subscribe(
      (locale) => {
        this.calLocale = CalLocaleSwitch.getLocaleCalendarObj(locale);
        this.overnightParkingText = this.translation.translate('overnightParkingText');
        this.issuesReportedText = this.translation.translate('issuesReportedText');
        this.poolParkingViolationText = this.translation.translate('poolParkingViolationText');
        this.vipParkingViolationText = this.translation.translate('vipParkingViolationText');
      }
    );
    this.overviewSub = this.reportsService.getVehicleParkingOverview()
      .subscribe(vehicleParkingOverview => {
        this.vehicleParkingOverview = vehicleParkingOverview;
      });

    this.vehicleDataSub = this.reportsService.getVehicleTrackData()
      .subscribe(vehicleTrackData => {
        this.vehTrackData = vehicleTrackData;
      });
  }

  ngOnDestroy() {
    if (this.vehicleDataSub)
      this.vehicleDataSub.unsubscribe();
    if (this.overviewSub)
      this.overviewSub.unsubscribe();
  }

}
