import { Component, OnInit, OnDestroy } from '@angular/core';
import { environment } from 'src/environments/environment';
import { ReportsService } from '../../../../services/reports.service';

import {
  Language,
  DefaultLocale,
  LocaleService,
  Timezone,
  TranslationService
} from 'angular-l10n';
import CalLocaleSwitch from 'src/app/shared/utils/cal-locale-switch';

@Component({
  selector: 'app-cab-tracking',
  templateUrl: './cab-tracking.component.html',
  styleUrls: ['./cab-tracking.component.scss']
})
export class CabTrackingComponent implements OnInit, OnDestroy {

  @Language() lang: string;
  @DefaultLocale() defaultLocale: string;
  @Timezone() timezone: string;
  calLocale: Object;
  issuesReportedText: string;
  cabsWithEscortText: string;
  womenPassengersText: string;
  tripsCompletedText: string;
  /**
   * cab tracking table data
   */
  cabData: any;
  /**
   * trip status data for donut chart
   */
  tripStatusData: Object[];
  /**
   * cab type data for donut chart
   */
  cabTypeDistrData: Object[];
  /**
   * date range to display the data
   */
  dateRange: Date[];

  // Adding the chart variables for the tripStatus
  tripStatusConstants_rightAlign: boolean = environment.tripStatus.CHARTPROPERTIES.RIGHTALIGN;;
  tripStatusConstants_chartid: String = environment.tripStatus.CHARTPROPERTIES.CHARTID;
  tripStatusConstants_addClearBtn: boolean = environment.tripStatus.CHARTPROPERTIES.ADDCLEARBTN;
  tripStatusConstants_legendColors: any = environment.tripStatus.LEGENDCOLORS;
  tripStatusConstants_options: any;

  // Adding the chart variables for the cabtype distribution
  cabTypeDistr_rightAlign: boolean = environment.cabTypeDistr.CHARTPROPERTIES.RIGHTALIGN;;
  cabTypeDistr_chartid: String = environment.cabTypeDistr.CHARTPROPERTIES.CHARTID;
  cabTypeDistr_addClearBtn: boolean = environment.cabTypeDistr.CHARTPROPERTIES.ADDCLEARBTN;
  cabTypeDistr_legendColors: any = environment.cabTypeDistr.LEGENDCOLORS;
  cabTypeDistr_options: any;

  cabModal: any = {
    opencabModal: false,
  };

  cabDataSub: any;
  /**
   * Constructor function used to initialse the cab tracking component
   * @param reportsService 
   */
  constructor(private reportsService: ReportsService, public locale: LocaleService, public translation: TranslationService) { }

  /**
   * shows cab tracking report generating modal
   */
  openCabModal() {
    this.cabModal.opencabModal = true;
  }

  /**
   * Init function is used to fetch the data required for cab tracking tab
   */
  ngOnInit() {
    this.tripStatusConstants_options = environment.tripStatus.CHARTPROPERTIES;
    this.cabTypeDistr_options = environment.cabTypeDistr.CHARTPROPERTIES;
    this.tripStatusData = [{ "key": "Trips Completed", "y": "400" }, { "key": "Trips Pending", "y": "100" }];
    this.cabTypeDistrData = [{ "key": "Cabs with Escort", "y": "400" }, { "key": "Cabs without Escort", "y": "100" }];

    this.translation.translationChanged().subscribe(
      (locale) => {
        this.calLocale = CalLocaleSwitch.getLocaleCalendarObj(locale);
        this.tripsCompletedText = this.translation.translate('tripsCompletedText');
        this.cabsWithEscortText = this.translation.translate('cabsWithEscortText');
        this.issuesReportedText = this.translation.translate('issuesReportedText');
        this.womenPassengersText = this.translation.translate('womenPassengersText');
      }
    );

    this.cabDataSub = this.reportsService.getCabData()
      .subscribe(cabData => {
        this.cabData = cabData;
      });
  }

  ngOnDestroy() {
    if (this.cabDataSub)
      this.cabDataSub.unsubscribe();
  }

}
