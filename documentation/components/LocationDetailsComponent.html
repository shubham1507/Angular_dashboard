import { Component, OnInit, ViewEncapsulation, ViewChild, OnDestroy } from '@angular/core';
import { environment } from 'src/environments/environment';
import { ReportsService } from '../../services/reports.service';
import * as moment from 'moment';
import { ActivatedRoute } from '@angular/router';
import { FormControl, FormGroup, FormBuilder, Validators } from '@angular/forms';
import { Subject, forkJoin } from 'rxjs';

import {
  Language,
  DefaultLocale,
  LocaleService,
  TranslationService
} from 'angular-l10n';

@Component({
  selector: 'physicalSecurity-location-details',
  templateUrl: './location-details.component.html',
  styleUrls: ['./location-details.component.scss'],
  encapsulation: ViewEncapsulation.None
})
export class LocationDetailsComponent implements OnInit, OnDestroy {

  @Language() lang: string;
  @DefaultLocale() defaultLocale: string;

  overviewActive: boolean = false;
  staffActive: boolean = false;
  reportsActive: boolean = false;
  openLocDropdown: boolean = false;

  officeLocations: any;


  calendarValue: Array<Date> = [moment().startOf('week').toDate(), moment().endOf('week').toDate()];

  invalidDates: Array<Date> = [];
  monthlyRoasterVal: any;

  locSub: any;
  selectedRouteParam: string;
  selectedOfficeLocation: string = 'Kalyani Vista  Bangalore';



  openBriefingModalSubject: Subject<any> = new Subject();

  openBroadcastModalSubject: Subject<any> = new Subject();

  notificationSubject: Subject<any> = new Subject();

  hideBadge: boolean = true;

  notificationCount: Number;

  searchtext : string;

  selectedOfcLocId : string = 'Kalyani_Vista_Bangalore';

  /**
   * Constructor function used to initialse the location details component
   * @param reportsService
   * @param route
   * @param formBuilder
   */
  constructor(public locale: LocaleService, public translation: TranslationService, private reportsService: ReportsService, private route: ActivatedRoute, private formBuilder: FormBuilder) { }

  /**
   * opens the briefing modal
   */
  openBriefingModal() {
    this.openBriefingModalSubject.next();
  }

  /**
   * opens the broadcast modal
   */
  openBroadcastModal() {
    this.openBroadcastModalSubject.next();
  }

  /**
   * shows the location dropdown
   * @param evt
   */
  showLocationMenu(evt) {
    evt.stopPropagation();
    this.openLocDropdown = true;
  }

  /**
   * selects office location
   * @param evt
   */
  selectOfcLocation(evt) {
    evt.stopPropagation();
    this.selectedOfficeLocation = this.translation.translate(evt.currentTarget.id);
    this.selectedOfcLocId = evt.currentTarget.id;
    this.openLocDropdown = false;
  }

  /**
   * hides the location dropdown
   */
  hideLocationMenu() {
    this.openLocDropdown = false;
  }

  /**
   * gets the notification count
   * @param notifCount notification count
   */
  getNotificationsCount(notifCount) {
    this.notificationCount = notifCount;
  }

  /**
   * shows notifications
   * @param evt
   */
  showNotifications(evt) {
    evt.stopPropagation();
    this.hideBadge = false;
    this.notificationSubject.next(true);
  }

  /**
   * hides notifications
   */
  hideNotifications() {
    this.notificationSubject.next(false);
  }

  /**
   * Init functions used to get the location name as route parameter
   */
  ngOnInit() {

    this.staffActive = true;
    this.locSub = this.route.paramMap.subscribe(paramsMap => {
      this.selectedRouteParam = paramsMap.get('selectedParam');
      this.selectedOfficeLocation = paramsMap.get('selectedParam');

      if (this.selectedRouteParam == "Reports") { //Add multiple conditions to set the route active for different modules

        console.log(this.reportsService.getSharedDTO()); // fetching the search result to be loaded onto the tabs

        this.staffActive = true;
        //this.attendanceActive =  true;
      }
    });

    this.reportsService.getOfficeLocation()
      .subscribe(officeLocations => {
        this.officeLocations = officeLocations;
      });

      this.translation.translationChanged().subscribe(
        () => {
          this.searchtext = this.translation.translate('searchtext');
          if(typeof(this.selectedOfcLocId)!=undefined){
            this.selectedOfficeLocation = this.translation.translate(this.selectedOfcLocId);
          }

        }
      );

      this.translation.translationError.subscribe((error) => {
          if (error) {
              console.log(error);
          }
      });

  }

  /**
   * destroy function used to unsubscribe all the subscription in this component
   */
  ngOnDestroy() {
    this.locSub.unsubscribe();
  }
}
