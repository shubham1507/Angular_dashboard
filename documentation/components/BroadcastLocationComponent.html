import { Component, OnInit, Input, ViewEncapsulation, OnDestroy } from '@angular/core';
import { FormGroup, FormControl , FormBuilder, Validators} from '@angular/forms';
import { Subject } from 'rxjs';
import { ReportsService } from 'src/app/services/reports.service';


import {
  Language,
  DefaultLocale,
  Timezone,
  TranslationService,
  LocaleService,
} from 'angular-l10n';


/**
 * Broadcast component used to broadcast the message for the provided office location
 */
@Component({
  selector: 'physicalsecurity-broadcast-location',
  templateUrl: './broadcast-location.component.html',
  styleUrls: ['./broadcast-location.component.scss'],
  encapsulation: ViewEncapsulation.None
})
export class BroadCastLocationComponent implements OnInit , OnDestroy {

  @Language() lang: string;
  @DefaultLocale() defaultLocale: string;

  autoCompleteText : string =  'Add Office Location';
  writeHere :  string  =  'WriteHere';


  today: Date = new Date();

  /**
   * Conditional flag used to open and close the broadcasr modal
   */
  locBroadCastFlag : boolean = false;

  /**
   * Place holder used to store the filtered locations
   */
  filteredLocations: any[];

  /**
   *Place holder used to store the location data
   */
  locationData: any = {};

  autocompletetext: string;
  writeheretext: string;


  /**
   * Subject for updating the state for activating the broadcast modal
   */
  @Input() openBroadcastModalSubject: Subject<any>;


  /**
   * Reactive broadcast form for capturing the office locaion and the description text
   */
  broadCastForm = new FormGroup ({
    officeLocation: new FormControl(),
    descText: new FormControl(),

  });


  /**
   * Function acts as complete call back when the auto search completes for the location input
   * @param event
   */
  filterLocationComplete(event) {
    let query = event.query;
    this.filteredLocations = this.filterLocation(query, this.locationData);
  }

  /**
   * Auto complete function used to filter the location based on the query provided
   * @param query
   * @param locations
   */
  filterLocation(query, locations: any[]): any[] {
    //in a real application, make a request to a remote url with the query and return filtered results, for demo we filter at client side
    let filtered: any[] = [];
    for (let i = 0; i < locations.length; i++) {
      let location = locations[i];
      if (location.name.toLowerCase().indexOf(query.toLowerCase()) == 0) {
        filtered.push(location);
      }
    }
    return filtered;
  }

  /**
   * On submit function when user is ready to submit the broadcasr form data
   */
  onSubmit() {
    // TODO: use service and do the api call for broadcast form details
    this.broadCastForm.reset();
  }


  /**
   * Function used to open the broadcast modal
   * set the initial broadcast form values and add default validations
   * subscribe to the value changes for the form properties and open the modal
   */
  openBroadCastModal() {
    this.broadCastForm = this.formBuilder.group({
      officeLocation: ['',Validators.required],
      descText: ['',Validators.required]
    });
    this.broadCastForm.valueChanges.subscribe(console.log);
    this.locBroadCastFlag = true;
  }


  /**
   * Constructor used to initalise the broadcast modal
   * @param formBuilder
   * @param reportsService
   */
  constructor(private formBuilder : FormBuilder, private reportsService: ReportsService , public translation: TranslationService , public locale : LocaleService) { }

  /**
   * Subscring to get the location and activation for the broadcast modal open operations
   */
  ngOnInit() {
    this.openBroadcastModalSubject.subscribe(v => {
      this.openBroadCastModal();
    });

    this.reportsService.getCountriesData()
    .subscribe(locationData => {
      this.locationData =  locationData.data;
     });

     this.translation.translationChanged().subscribe(
      () => {
        this.autocompletetext = this.translation.translate('broadcast.autocompletetext') ;
        this.writeheretext = this.translation.translate('broadcast.writehere') ;

       });
  }

  ngOnDestroy(){

  }
}
