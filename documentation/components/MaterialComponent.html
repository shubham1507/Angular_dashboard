import { Component, OnInit, OnDestroy } from '@angular/core';
import { environment } from 'src/environments/environment';
import { ReportsService } from '../../../../services/reports.service';
import { Subject, forkJoin } from 'rxjs';
import { map } from 'rxjs/operators';
import { MaterialReportData, MaterialLogCountData } from '../../material-api-models';
import { FormControl, FormGroup, FormBuilder, Validators } from '@angular/forms';
import { ClrDatagridStateInterface } from "@clr/angular";
import { saveAs } from 'file-saver';

import {
  Language,
  DefaultLocale,
  LocaleService,
  Timezone,
  TranslationService
} from 'angular-l10n';
import CalLocaleSwitch from 'src/app/shared/utils/cal-locale-switch';

@Component({
  selector: 'app-material',
  templateUrl: './material.component.html',
  styleUrls: ['./material.component.scss']
})
export class MaterialComponent implements OnInit, OnDestroy {

  @Language() lang: string;
  @DefaultLocale() defaultLocale: string;
  @Timezone() timezone: string;
  calLocale: Object;
  issuesReportedText: string;
  incompleteTransText: string;
  outwardMatEntriesText: string;
  inwardMatEntriesText: string;
  /**
   * material table data
   */
  materialData: any;
  /**
   * material tab overview data(counts)
   */
  materialOverviewData = {};
  /**
   * material transaction status data for donut chart
   */
  matTransStatusData: Object[];
  /**
   * material transaction type data for donut chart
   */
  matTransTypeData: Object[];

  /**
   * selected movement type
   */
  selectedMovementType: string = '';
  /**
   * selected entry type
   */
  selectedEntryType: string = '';
  /**
   * material table state (contains sorting, filter details)
   */
  materialTableState: any;
  /**
   * page no of current page
   */
  materialTablePageNo = 1;
  /**
   * flag to show/hide loading spinner
   */
  materialTableLoading: boolean = false;
  /**
   * selected value from dropdown
   */
  selectedFilterValue = 'Filter By Type';
  /**
   * selected value from dropdown
   */
  selectedFilterValue1 = 'Filter By Entry Type';
  /**
   * date range for the material data to be dispayed
   */
  materialDateRange: Date[];
  /**
   * start time for the data in the report
   */
  materialReportStartTime: Date;
  /**
   * end time for the data in the report
   */
  materialReportEndTime: Date;

   * material report form
   */
  newMaterialForm: FormGroup;

  matTransStatusConstant_chartid: String = environment.matTransStatus.CHARTPROPERTIES.CHARTID;
  matTransStatusConstant_addClearBtn: boolean = environment.matTransStatus.CHARTPROPERTIES.ADDCLEARBTN;
  matTransStatusConstant_legendColors: any = environment.matTransStatus.LEGENDCOLORS;

  matTransTypeConstant_rightAlign: boolean = environment.matTransType.CHARTPROPERTIES.RIGHTALIGN;;
  matTransTypeConstant_chartid: String = environment.matTransType.CHARTPROPERTIES.CHARTID;
  matTransTypeConstant_addClearBtn: boolean = environment.matTransType.CHARTPROPERTIES.ADDCLEARBTN;
  matTransTypeConstant_legendColors: any = environment.matTransType.LEGENDCOLORS;
  matTransTypeConstant_options: any;

  materialModal: any = {
    openmaterialModal: false,
  };

  materialDataSub: any;
  materialOverviewSub: any;

  /**
   * Constructor function used to initialse the material component
   * @param reportsService 
   * @param formBuilder 
   */
  constructor(private reportsService: ReportsService, private formBuilder: FormBuilder, public locale: LocaleService, public translation: TranslationService) { }

  /**
   * shows material report generating modal
   */
  openMaterialModal() {
    this.materialModal.openmaterialModal = true;
  }

  /**
   * refreshes the table data when state changes
   * @param state 
   */
  refresh(state: ClrDatagridStateInterface) {
    this.materialTableState = state;
    this.updateMaterialData();
  }

  /**
   * submits the request for generating the report in xls format
   */
  onMaterialSubmit() {
    let requestBody: Object = {};
    requestBody['startTime'] = this.newMaterialForm.value['startTime'].getTime();
    requestBody['endTime'] = this.newMaterialForm.value['endTime'].getTime();
    requestBody['reportRequestorEmailId'] = "jgupte@vmware.com";
    if (this.newMaterialForm.value['movementType'] != "All") {
      requestBody['materialMovementType'] = this.newMaterialForm.value['movementType'];
    }
    else {
      requestBody['materialMovementType'] = null;
    }
    if (this.newMaterialForm.value['itemType'] != "All") {
      requestBody['itemType'] = this.newMaterialForm.value['itemType'];
    }
    else {
      requestBody['itemType'] = null;
    }
    this.reportsService.getMaterialReport(requestBody)
      .subscribe(data => {
        saveAs(data.body, "material-report.xls");
      },
        error => {
          console.log("error", error);
        })
  }

  /**
   * selects filter movement type
   * @param movementType 
   * @param evt 
   */
  selectMovementType(movementType: string, evt) {
    this.selectedFilterValue = evt.currentTarget.innerText;
    this.selectedMovementType = movementType;
    this.updateMaterialData();
  }

  /**
   * selects filter entry type
   * @param entryType 
   * @param evt 
   */
  selectEntryType(entryType: string, evt) {
    this.selectedFilterValue1 = evt.currentTarget.innerText;
    this.selectedEntryType = entryType;
    this.updateMaterialData();
  }

  /**
   * selects date range filter
   * @param evt 
   */
  selectDateRange(evt) {
    if (this.materialDateRange[0] && this.materialDateRange[1]) {
      this.updateMaterialData();
    }
  }

  /**
   * updates material table data when any filter or sorting is changed
   */
  updateMaterialData() {

    // As per the  clarity issue  : https://github.com/vmware/clarity/issues/1519 explained happens only for mock data
    //TODO: We added timeout as a work around for loading flag , need to validate and remove at the time of Api Integration
    setTimeout(() => { this.materialTableLoading = true; }, 0);


    let materialRequestBody = {};
    if (this.selectedMovementType)
      materialRequestBody['materialMovementType'] = this.selectedMovementType;
    if (this.selectedEntryType)
      materialRequestBody['itemType'] = this.selectedEntryType;
    materialRequestBody['pageNumber'] = this.materialTablePageNo;
    materialRequestBody['pageSize'] = 10;
    if (this.materialTableState && this.materialTableState.sort) {
      materialRequestBody['sortBy'] = this.materialTableState.sort.by;
      materialRequestBody['reverse'] = this.materialTableState.sort.reverse;
    }
    if (this.materialDateRange && this.materialDateRange[0]) {
      materialRequestBody['startTime'] = this.materialDateRange[0].getTime();
    }
    if (this.materialDateRange && this.materialDateRange[1]) {
      materialRequestBody['endTime'] = this.materialDateRange[1].getTime();
    }
    this.materialDataSub = this.reportsService.getMaterialData(materialRequestBody)
      .pipe(map((materialData: MaterialReportData) => {
        setTimeout(() => {
          this.materialData = materialData;
          this.materialTableLoading = false;
        }, 0)
      }))
      .subscribe(res => { })
  }

  /**
   * * Init function is used to fetch the initial data required for material tab
   */
  ngOnInit() {
    this.matTransStatusConstant_options = environment.matTransStatus.CHARTPROPERTIES;
    this.matTransTypeConstant_options = environment.matTransType.CHARTPROPERTIES;

    this.translation.translationChanged().subscribe(
      (locale) => {
        this.calLocale = CalLocaleSwitch.getLocaleCalendarObj(locale);
        this.incompleteTransText = this.translation.translate('incompleteTransText');
        this.inwardMatEntriesText = this.translation.translate('inwardMatEntriesText');
        this.issuesReportedText = this.translation.translate('issuesReportedText');
        this.outwardMatEntriesText = this.translation.translate('outwardMatEntriesText');
      }
    );
    let inwardLog = this.reportsService.getMaterialLogCount("inward");
    let outwardLog = this.reportsService.getMaterialLogCount("outward");
    let returnableLog = this.reportsService.getMaterialLogCount("returnable");
    let nonReturnableLog = this.reportsService.getMaterialLogCount("nonreturnable");
    let completeLog = this.reportsService.getMaterialLogCount("complete");
    let incompleteLog = this.reportsService.getMaterialLogCount("incomplete");
    let issuesReportedLog = this.reportsService.getMaterialLogCount("issuereported");

    this.materialOverviewSub = forkJoin([inwardLog, outwardLog, returnableLog, nonReturnableLog, completeLog, incompleteLog, issuesReportedLog])
      .pipe(map((response: MaterialLogCountData[]) => {
        this.materialOverviewData["inwardLogCount"] = response[0].data.count;
        this.materialOverviewData["outwardLogCount"] = response[1].data.count;
        this.materialOverviewData["returnableLogCount"] = response[2].data.count;
        this.materialOverviewData["nonReturnableLogCount"] = response[3].data.count;
        this.materialOverviewData["completeTransLogCount"] = response[4].data.count;
        this.materialOverviewData["incompleteTransLogCount"] = response[5].data.count;
        this.materialOverviewData["issuesLogCount"] = response[6].data.count;

        this.matTransStatusData = [{ "key": "Incomplete Transactions", "y": this.materialOverviewData["incompleteTransLogCount"] }, { "key": "Complete Transactions", "y": this.materialOverviewData["completeTransLogCount"] }];
        this.matTransTypeData = [{ "key": "Returnable", "y": this.materialOverviewData["returnableLogCount"] }, { "key": "Non-Returnable", "y": this.materialOverviewData["nonReturnableLogCount"] }];
      }))
      .subscribe(res => { });
    this.updateMaterialData();

    let currentDate = new Date().setHours(0, 0, 0, 0);
    let currentDateLastMinute = new Date().setHours(23, 59, 0, 0);
    this.materialDateRange = [new Date(currentDate), null];
    this.materialReportStartTime = new Date(currentDate);
    this.materialReportEndTime = new Date(currentDateLastMinute);

    this.newMaterialForm = new FormGroup({
      movementType: new FormControl('All'),
      itemType: new FormControl('All'),
      startTime: new FormControl(this.materialReportStartTime),
      endTime: new FormControl(this.materialReportEndTime)
    });
  }

  ngOnDestroy() {
    if (this.materialDataSub)
      this.materialDataSub.unsubscribe();
    if (this.materialOverviewSub)
      this.materialOverviewSub.unsubscribe();
  }

}
