import { Component, OnInit, OnDestroy } from '@angular/core';
import { environment } from 'src/environments/environment';
import { ReportsService } from '../../../../services/reports.service';

import {
  Language,
  DefaultLocale,
  LocaleService,
  Timezone,
  TranslationService
} from 'angular-l10n';
import CalLocaleSwitch from 'src/app/shared/utils/cal-locale-switch';

@Component({
  selector: 'app-attendance',
  templateUrl: './attendance.component.html',
  styleUrls: ['./attendance.component.scss']
})
export class AttendanceComponent implements OnInit, OnDestroy {

  @Language() lang: string;
  @DefaultLocale() defaultLocale: string;
  @Timezone() timezone: string;
  calLocale: Object;
  allStaffText: string;
  staffOnLeaveText: string;
  femaleStaffText: string;
  incidentsReportedText: string;
  /**
   * attendance data displayed in table
   */
  attendaceData: any;
  /**
   * rolewise distribution data for donut chart
   */
  rateWiseDistrData: Object[];
  /**
   * demographic distribution data for donut chart
   */
  demoGraphDistrData: Object[];
  /**
   * date range to display the data
   */
  dateRange: Date[];

  // Adding the chart variables for the rate wise distribuition
  rateWiseDistrConstant_rightAlign: boolean = environment.rateWiseDistr.CHARTPROPERTIES.RIGHTALIGN;;
  rateWiseDistrConstant_chartid: String = environment.rateWiseDistr.CHARTPROPERTIES.CHARTID;
  rateWiseDistrConstant_addClearBtn: boolean = environment.rateWiseDistr.CHARTPROPERTIES.ADDCLEARBTN;
  rateWiseDistrConstant_legendColors: any = environment.rateWiseDistr.LEGENDCOLORS;
  rateWiseDistrConstant_options: any;


  // Adding the chart variables for the rate wise distribuition
  demoGraphDistrConstant_rightAlign: boolean = environment.demoGraphDistr.CHARTPROPERTIES.RIGHTALIGN;;
  demoGraphDistrConstant_chartid: String = environment.demoGraphDistr.CHARTPROPERTIES.CHARTID;
  demoGraphDistrConstant_addClearBtn: boolean = environment.demoGraphDistr.CHARTPROPERTIES.ADDCLEARBTN;
  demoGraphDistrConstant_legendColors: any = environment.demoGraphDistr.LEGENDCOLORS;
  demoGraphDistrConstant_options: any;

  attendanceModal: any = {
    openattendanceModal: false,
  };

  attendanceSub: any;
  /**
   * Constructor function used to initialse the attendance component
   * @param reportsService 
   */
  constructor(private reportsService: ReportsService, public locale: LocaleService, public translation: TranslationService) { }

  /**
   * shows attendance report generating modal
   */
  openAttendanceModal() {
    this.attendanceModal.openattendanceModal = true;
  }

  /**
   * Init function is used to fetch the data required for attendance tab
   */
  ngOnInit() {
    this.rateWiseDistrConstant_options = environment.rateWiseDistr.CHARTPROPERTIES;
    this.demoGraphDistrConstant_options = environment.demoGraphDistr.CHARTPROPERTIES;
    this.rateWiseDistrData = [{ "key": "Security Guards", "y": "453" }, { "key": "Senior Security Guards", "y": "345" }, { "key": "Security Executives", "y": "115" }, { "key": "Senior Security Executives", "y": "103" }];
    this.demoGraphDistrData = [{ "key": "North Zone", "y": "453" }, { "key": "East Zone", "y": "345" }, { "key": "West Zone", "y": "200" }, { "key": "South Zone", "y": "300" }];

    this.translation.translationChanged().subscribe(
      (locale) => {
        this.calLocale = CalLocaleSwitch.getLocaleCalendarObj(locale);
        this.allStaffText = this.translation.translate('allStaffText');
        this.femaleStaffText = this.translation.translate('femaleStaffText');
        this.incidentsReportedText = this.translation.translate('incidentsReportedText');
        this.staffOnLeaveText = this.translation.translate('staffOnLeaveText');
      }
    );

    this.attendanceSub = this.reportsService.getAttendanceData()
      .subscribe(attendaceData => {
        this.attendaceData = attendaceData;
      });
  }

  ngOnDestroy() {
    if (this.attendanceSub)
      this.attendanceSub.unsubscribe();
  }

}
