import { Component, OnInit, OnDestroy, ViewEncapsulation, ChangeDetectionStrategy, ChangeDetectorRef } from '@angular/core';
import { PeopleFinderService } from 'src/app/services/people-finder.service';
import { Subject } from 'rxjs';
import { ReportsService } from 'src/app/services/reports.service';
import { environment } from 'src/environments/environment';

import {
  Language,
  DefaultLocale,
  LocaleService,
  TranslationService
} from 'angular-l10n';


/**
 * dashboard component for the dashboard view
 * includes the consolidated view for the eventtime line
 * news feed incident data
 */
@Component({
  selector: 'physicalSecurity-dashboard',
  templateUrl: './dashboard.component.html',
  styleUrls: ['./dashboard.component.scss'],
  encapsulation: ViewEncapsulation.None
})
export class DashboardComponent implements OnInit, OnDestroy {

  @Language() lang: string;
  @DefaultLocale() defaultLocale: string;
  welcomeText: string;
  openIncidentsText: string;
  resourceShortageText: string;
  staffCompTasksText: string;
  newsFeedText: string;
  /**
   * TODO: Need to add documentation for place holders with actual purpose after
   * data integration
   */

  /**
   * place holder for storing incidents data
   */
  incidentsData: any;

  /**
   * place holder for storing the incident Type list
   */
  incidentTypeList: any[] = [];

  /**
   * Place holder for storing the region list
   */
  regionList = [];

  /**
   * conditional flag for opening the incident details modal
   */
  incidentDetailsModal: boolean = false;

  /**
   * place holder for storing the staff Compliance data
   */
  staffComplianceData: any;

  /**
  * Array Place holder for storing the compliance data regions
   */
  complianceDataRegions: Object[];


  /**
   * place holder for storing the resource shortage data
   */
  resourceShortageData: any;

  /**
   * place holder for storing the regions
   */
  regions = ["All Regions", "APAC", "AMER", "EMEA", "INDIA", "PALO ALTO"];

  /**
   * place holder for storing the user data
   */
  userData: any;

  /**
   * Conditional subject instance for opening the briefing modal
   */
  openBriefingModalSubject: Subject<any> = new Subject();

  /**
   * Conditional subject instance for opening the notifications
   */
  notificationSubject: Subject<any> = new Subject();

  /**
   * Conditional subject instance for opening the broadcast modal
   */
  openBroadcastModalSubject: Subject<any> = new Subject();

  /**
   * place holder for storing the notifications count
   */
  notificationCount: Number;

  /**
   * conditional flag for showing and hiding the notification badge count
   */
  hideBadge: boolean = true;

  addClearBtn: boolean = environment.incConstants.CHARTPROPERTIES.ADDCLEARBTN;
  rightAlign: boolean = environment.incConstants.CHARTPROPERTIES.RIGHTALIGN;
  options: any;
  clearText = environment.incConstants.CLEAR;;
  legendColors = environment.incConstants.LEGENDCOLORS;
  chartid = environment.incConstants.CHARTPROPERTIES.CHARTID;

  incidentSub: any;
  complianceSub: any;
  shortageSub: any;
  filteredIncidentSub: any;
  pfAccessTokenSub: any;
  peopleFinderSub: any;

  /***
   * Constructor function used to initialise the dashboard component
   * @param reportsService
   * @param peopleFinderService
   */
  constructor(public locale: LocaleService,
    public translation: TranslationService,
    private reportsService: ReportsService,
    private peopleFinderService: PeopleFinderService) { }


  /**
   * open the briefing modal component from dashboard
   */
  openBriefingModal() {
    this.openBriefingModalSubject.next();
  }

  /**
    * open the  broadcasr modal component from dashboard
    */
  openBroadcastModal() {
    this.openBroadcastModalSubject.next();
  }



  /**
    * get the total notifications count for the notifications
    */
  getNotificationsCount(notifCount) {
    this.notificationCount = notifCount;
  }

  /**
   * Show the notification messages for the logged in user
   * @param evt
   */
  showNotifications(evt) {
    evt.stopPropagation();
    this.hideBadge = false;
    this.notificationSubject.next(true);
  }

  /**
   * Hide the notifications when user clicks out side the notifications container
   */
  hideNotifications() {
    this.notificationSubject.next(false);
  }


  /**
   *clear the incident and reset the chart type data
   * for the incidents
   */
  clearFunction() {
    this.reportsService.getIncidents()
      .subscribe(incidentsData => {
        this.incidentTypeList = [];
        this.regionList = Object.keys(incidentsData.incidentsRegionWise);
        for (let incidentType in incidentsData.incidentTypes) {
          this.incidentTypeList.push({
            key: incidentType,
            y: incidentsData.incidentTypes[incidentType]
          });
        }
        this.incidentsData = incidentsData;
      }
      )
  }

  /**
   * Populated the chart data for the selected incident type
   * @param selectedIncidentType
   */
  legendClick(selectedIncidentType: string) {
    this.filteredIncidentSub = this.reportsService.getFilteredIncidents()
      .subscribe(incidentsData => {
        this.incidentTypeList = [];
        this.regionList = Object.keys(incidentsData.incidentsRegionWise);
        for (let incidentType in incidentsData.incidentTypes) {
          this.incidentTypeList.push({
            key: incidentType,
            y: incidentsData.incidentTypes[incidentType]
          });
        }
        this.incidentsData = incidentsData;
      }
      )
  }

  /**
   * open the incident modal
   * @param incident
   */
  openIncidentModal(incident: any) {
    this.incidentDetailsModal = true;
  }

  /**
   * Get the user data from the people finder access token
   */
  getUserDataFromPeopleFinder() {
    this.pfAccessTokenSub = this.peopleFinderService.getPeopleFinderAccessToken()
      .subscribe((response: any) => {
        if (typeof response == "object") {
          this.peopleFinderService.setPeopleFinderTokenInLocalStorage(response);
        }
        //hardcoded person ID has to be removed
        this.peopleFinderSub = this.peopleFinderService.getPersonData("rchetry")
          .subscribe((userData: any) => {
            if (userData)
              this.userData = userData.users[0];
          })
      });
  }

  /**
   * Get the user data for the people finder
   * incident data and compliance data
   */
  ngOnInit() {
    this.options = environment.incConstants.CHARTPROPERTIES;
    this.getUserDataFromPeopleFinder();

    this.incidentSub = this.reportsService.getIncidents()
      .subscribe(incidentsData => {
        this.regionList = Object.keys(incidentsData.incidentsRegionWise);
        for (let incidentType in incidentsData.incidentTypes) {
          this.incidentTypeList.push({
            key: incidentType,
            y: incidentsData.incidentTypes[incidentType]
          });
        }
        this.incidentsData = incidentsData;
      }
      )

    this.complianceSub = this.reportsService.getComplianceData()
      .subscribe(staffData => {
        this.staffComplianceData = staffData;
        this.complianceDataRegions = Object.keys(staffData.complianceRegions);
      });

    this.shortageSub = this.reportsService.getShortageData()
      .subscribe(response => {
        this.resourceShortageData = response;
      })

    this.translation.translationChanged().subscribe(
      () => {
        this.welcomeText = this.translation.translate('welcomeText');
        this.openIncidentsText=this.translation.translate('openIncidentsText');
        this.resourceShortageText=this.translation.translate('resourceShortageText');
        this.staffCompTasksText=this.translation.translate('staffCompTasksText');
        this.newsFeedText=this.translation.translate('newsFeedText');
      }
    );

    this.translation.translationError.subscribe((error) => {
      if (error) {
        console.log(error);
      }
    });

  }

  /**
   * Unsubscribe all the service methods
   */
  ngOnDestroy() {
    if (this.incidentSub)
      this.incidentSub.unsubscribe();
    if (this.complianceSub)
      this.complianceSub.unsubscribe();
    if (this.shortageSub)
      this.shortageSub.unsubscribe();
    if (this.filteredIncidentSub)
      this.filteredIncidentSub.unsubscribe();
    if (this.peopleFinderSub)
      this.peopleFinderSub.unsubscribe();
    if (this.pfAccessTokenSub)
      this.pfAccessTokenSub.unsubscribe();
  }
}
